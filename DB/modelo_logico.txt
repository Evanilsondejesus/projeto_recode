



-- Crie a tabela
CREATE TABLE TabelaExemplo (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    Coluna1 VARCHAR(50),
    Coluna2 VARCHAR(50),
    Coluna3 INT
);

-- Crie um gatilho para preencher as colunas automaticamente
DELIMITER //
CREATE TRIGGER PreencherColunasAutomaticamente
BEFORE INSERT ON TabelaExemplo
FOR EACH ROW
BEGIN
    -- Verifique se a coluna Coluna1 está sendo inserida com um valor específico
    IF NEW.Coluna1 = 'ValorEspecifico' THEN
        -- Preencha as colunas Coluna2 e Coluna3 automaticamente
        SET NEW.Coluna2 = 'ValorParaColuna2';
        SET NEW.Coluna3 = 42; -- Um valor numérico de exemplo
    END IF;
END;
//
DELIMITER ;

-- Crie a tabela "Usuário" (TabelaReferenciada)


CREATE TABLE Usuario (
    UsuarioID INT PRIMARY KEY AUTO_INCREMENT,
    Email VARCHAR(255) NOT NULL,
    Senha VARCHAR(255) NOT NULL
);






CREATE TABLE usuarios (
    Email VARCHAR(255) NOT NULL,
    Senha VARCHAR(255) NOT NULL,
    usuarioID INT PRIMARY KEY AUTO_INCREMENT,
);




CREATE TABLE pacotes(
    nome_pacote VARCHAR(255) NOT NULL,
    destino VARCHAR(255) NOT NULL,
    descontos VARCHAR(10) NOT NULL,
    valor INT NOT NULL,
    valor_total INT,
    id_pacote INT PRIMARY KEY AUTO_INCREMENT

);





CREATE TABLE Registros (
    RegistroID INT PRIMARY KEY AUTO_INCREMENT,
    usuarioID INT,
    id_pacote INT,


    Email VARCHAR(255),
    Senha VARCHAR(255),

    FOREIGN KEY (UsuarioID) REFERENCES usuarios(usuarioID),
    FOREIGN KEY (id_pacote) REFERENCES pacotes(id_pacote)

);
